# Root = From Market
# scope:market = Market merging into
# scope:trade_center = Trade Center state of root
on_merge_markets = {
    on_actions = {
        ase_on_merge_markets
    }
}

ase_on_merge_markets = {
    effect = {
        if = {
            limit = {
                exists = owner
            }
            owner = {
                trigger_event = {
                    id = ase_framework_events.3
                }
            }
        }
        scope:market = {
            if = {
                limit = {
                    exists = owner
                }
            }
            owner = {
                trigger_event = {
                    id = ase_framework_events.3
                }
            }
        }
    }
}

# Root = Country
on_rank_changed = {
	on_actions = {
        ase_on_rank_changed
    }
}

ase_on_rank_changed = {
    effect = {
        if = {
            limit = {
                NOT = {
                    is_country_type = decentralized
                }
                NOT = {
                    this = market.owner
                }
            }
            market.owner = {
                ase_recalculate_market_and_country_reserve = yes
            }
        }
    }
}
